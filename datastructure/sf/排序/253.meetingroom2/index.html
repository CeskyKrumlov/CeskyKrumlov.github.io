public int minMeetingRooms(int[][] intervals) {

// цМЙчЕзф╝ЪшоочЪДх╝АхзЛцЧ╢щЧ┤цОТх║П

// ф╕Нф╕║чй║я╝Мф╗ех╜УхЙНх╝АхзЛцЧ╢щЧ┤ цпФш╛Г хаЖщб╢я╝МхаЖщб╢хН│цЬАцЧйч╗УцЭЯцЧ╢щЧ┤

heap.add(intervals[i][1]);

public int minMeetingRooms(int[][] intervals) {

Arrays.sort(ends);
int bIndex = 0;

else eIndex++;
bIndex++;
}

}
